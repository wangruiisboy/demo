``` python

class Solution:
    def decimalRepresentation(self, n: int) -> List[int]:
        ans = []
        power10 = 1
        while n:
            n,mod = divmod(n,10)
            if 1<=mod<=9:
                ans.append(power10 * mod)
            power10 *= 10
        ans.reverse()
        return ans
```
