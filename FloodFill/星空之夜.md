
## 1.思路
（1）什么方法求连通块? DFS/BFS/并查集Union Find
（2）如何区分不同的连通块：可以使用哈希函数。O(1)判断
（3）如何设计哈希函数,按照题目的要求
 (4) 修改所有连通块中的字母

``` c++
#include<algorithm>
#include<cstring>
#include<cmath>
#include<iostream>
#define x first
#define y second

using namespace std;

//存二元坐标
typedef pair<int, int> PII;

const int N = 110;
const int M = 170;
int n,m;
//存储举证
char g[N][N];
PII q[M];
//表示每个星群有多少个星星，cnt表示当前存到了第几个哈希值(下标)
int top = 0,cnt = 0;

//最多26个，还是多开点
double hash_val[30];
double eps = 1e-8;

void dfs(int a,int b){
    q[top++] = {a,b};
    g[a][b] = '0';
    
    //遍历8个方向
    for(int x=a-1;x<=a+1;x++){
        for(int y = b-1;y<=b+1;y++){
            
            //没有越界，并且是星星
            if(x>=0 && x<n && y>=0 && y<m && g[x][y] =='1'){
                 dfs(x,y);
            }
           
        }
    }
}


//哈希函数：计算欧几里得距离
double get_dist(PII&a,PII&b){
    double dx = a.x - b.x,dy = a.y - b.y;
    return sqrt(dx*dx + dy * dy);
}


//所有点的欧几里得距离之和算出哈希值
double get_hash(){
    
    double sum = 0;
    //枚举所有星群的所有的星星，累加所有欧几里得距离
    for(int i=0;i<top;i++){
        for(int j=0;j<i;j++){
            sum+= get_dist(q[i],q[j]);
        }
    }
    return sum;
}
char get_id(){
    //就算当前的哈希值
    double val = get_hash();
    
    //是否已经有过哈希值了
    for(int i=0;i<cnt;i++){
        //高精度判断相等
        if(abs(hash_val[i] - val)  < eps){
            return 'a'+i;
        }
    }
    
    //没有就新开
    hash_val[cnt++] = val;
    return 'a'+cnt - 1;
}
int main()
{
    ios::sync_with_stdio(false),cin.tie(0),cout.tie(0);
    
    //1坑，先度入列
    cin >> m >> n;
    
    //读入举证
    for(int i=0;i<n;i++){
        cin >> g[i];
    }
    
    //遍历每个点，flood fill
    for(int i=0;i<n;i++){
        for(int j=0;j<m;j++){
            if(g[i][j] == '1'){
                //清空top
                top = 0;
                dfs(i,j);
                //获取哈希之后的位置
                char id = get_id();
                //将连通块内的所有点都改为id
                for(int k=0;k<top;k++){
                    g[q[k].x][q[k].y] = id;
                }
            }
        }
    }
    
    //遍历输出举证
    for(int i=0;i<n;i++){
        puts(g[i]);
    }
    
    return 0;
}
```
