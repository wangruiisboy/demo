class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        n,m = len(grid),len(grid[0])
        ans = 0
        def dfs(i,j):
            if (i,j) in vis:
                return 
            if i<0 or i>n-1 or j<0 or j>m-1:
                return
            if grid[i][j] == '0':
                return
            vis.add((i,j))
            
            dfs(i-1,j)
            dfs(i+1,j)
            dfs(i,j-1)
            dfs(i,j+1)
        
        # 遍历整个网络
        vis = set()
        for i in range(n):
            for j in range(m):
                if grid[i][j] =='1' and (i,j) not in vis:
                    ans+=1
                    dfs(i,j)
        return ans
